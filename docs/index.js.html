const { fakeApiCall } = require('./lib/http');

/**
 * calling multiple promises
 */
const allSettledTest = async () => {
  try {
    const t = await Promise.allSettled([fakeApiCall(1000), fakeApiCall(10)]);
    console.log(t);
  } catch (e) {
    console.error(e);
  }
};
